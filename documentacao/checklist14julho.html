Fase 1: Ajuste do Banco (SQL)

Copie e cole este bloco no SQL Editor do seu Supabase:

-- 1. Criar Tabela de Hóspedes
CREATE TABLE hospedes (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  pousada_id UUID REFERENCES pousadas(id),
  nome TEXT NOT NULL,
  cpf TEXT,
  email TEXT,
  telefone TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 2. Atualizar Tabela Reservas
ALTER TABLE reservas 
ADD COLUMN hospede_id UUID REFERENCES hospedes(id),
ADD COLUMN valor_sinal NUMERIC,
ADD COLUMN documento_url TEXT;

-- 3. Atualizar Tabela Pousadas
ALTER TABLE pousadas ADD COLUMN senha_gestor TEXT;
